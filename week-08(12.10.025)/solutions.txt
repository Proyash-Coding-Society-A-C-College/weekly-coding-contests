#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int num;
    cin >> num;

    while (num >= 10) {  
        int sum = 0;

        
        while (num > 0) {
            sum += num % 10;  
            num /= 10;        
        }

        num = sum; 
    }

    cout << num;
    return 0;
}
